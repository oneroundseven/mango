js: 

var Upload = FOCUS.widget.Upload;

Upload.plugins.myplugin = function(mode){
	this.on('ready', function(){
		alert('hello Uploader')
	})
};

var MODE = Upload.MODE;

var uploader = new Upload({
	priority: [ MODE.HTML5, MODE.FLASH, MODE.IFRAME ], // 上传核心的初始化优先顺序
	placeholder: '#file', // 占位元素
	uploadURL: 'file.do', // 上传的目标URL
	filePostName: 'docfile', // 上传时文件使用的 name
	fileTypes: '*.pdf, *.doc, *.docx', // 允许选择的文件类型
	description: 'Document Files', // 文件选择对话框上的 描述文字
	queueLimit: 3, // 队列长度限制
	sizeLimit: '1MB', // 单个文件尺寸限制
	postParams: { // 上传文件时携带的数据
		hello: 'world'
	},
	timeout: 5000, // 单次上传操作超时设置(ms)
	charset: 'UTF-8', // 上传使用的编码格式，不建议使用 UTF-8 以外的参数
	multiple: false, // 允许多选
	button: {
		text: 'Upload Docs' // 按钮显示的文字
	},
	plugins: 'myplugin', // 自定义插件列表
	debug: false // 调试开关
});
