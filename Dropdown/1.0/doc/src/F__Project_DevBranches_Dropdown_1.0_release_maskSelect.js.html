363                     cItem.className = this.config.style.tagName + '-' + data[i].type;
364                     container.appendChild(cItem);
365                 } else {
366                     cItem.innerHTML = data[i].name;
367                     container.appendChild(cItem);

369                     container.appendChild(arguments.callee.call(this, data[i].value, document.createDocumentFragment()));

375                 item.innerHTML = '<a href="javascript:;">'+ data[i].name +'</a>';
376             }
377 
378             dfrag.appendChild(item);

401     /**
402      * convert select options value to JSON
403      * @param {Element} node Select
404      * @param {Array} data
405      * @return {Array}
406      */
407     function select2Json(node, data) {
408         if (!node) {
409             return null;
410         }
411         var items = node.childNodes, i, item;
412 
413         for (i = 0; i < items.length; i++) {
414             if (items[i] && items[i].nodeType === 1) {
415                 if (items[i].nodeName.toLowerCase() === 'optgroup') {
416                     item = {
417                         name : items[i].label,
418                         value : arguments.callee(items[i], [])
419                     };
420 
421                     if (items[i].getAttributeNode('data-type')) {
422                         item.type = items[i].getAttributeNode('data-type').nodeValue;
423                     }
424                 } else {
425                     item = {
426                         name : items[i].text,
427                         value : items[i].value
428                     };
429                 }
430 
431                 data.push(item);
432             }
433         }
434 
435         return data;
436     }
