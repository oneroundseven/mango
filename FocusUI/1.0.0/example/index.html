//单独使用：			
FocusUI.cover.show({//遮罩
	bgColor: '#ffff00',//背景色
	bgOpacity: 0.4,	//透明度					
	showAnim:"fadeIn",//显示效果
	dialog:"#dialog1",//遮罩层上的div
	overlay:true,//多层遮罩之间颜色是否重叠效果
	baseFloatZIndex:100//基本z-index
});
FocusUI.center({//相对浏览器定位
	dialog:"#dialog1",//居中的div选择器或对象
	position:"fixed"//样式参数
	top: '50%',
	left: '50%',
	marginTop: 0,
	relative: true
});
FocusUI.drag({//拖拽
	dialog:"#dialog1",//移动的物件
	title:"#title1",//触发移动的物件
	margins:[0,0,0,0],//移动限制的边距
	magnetDistance,//与浏览器边缘吸附效果的范围值
	rtl:true//是否右对齐
});			
FocusUI.bgIframe({//ie6下select与div层次关系BUG处理
	dialog: dialog,//处理的div
	bgColor:"white"//iframe覆盖的背景色
});

//统一调用：
$("#dialog").focusUI({
	cover: true,
	drag: true,
	center: true,
	bgIframe:true,
    title:null
});	

//focusUI全局参数设置：
FocusUI.setConfig({
	X:X,
	Y:Y
});
